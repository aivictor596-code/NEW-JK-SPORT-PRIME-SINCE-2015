  const box=document.getElementById("products");

  current=cat;

function addCart(i){
  cart.push(products[i]);
  document.getElementById("count").innerText=cart.length;

  if(cart.length===0){

  window.open(
    `https://wa.me/${WHATSAPP_NUMBER}?text=${encodeURIComponent(msg)}`
  );
}
